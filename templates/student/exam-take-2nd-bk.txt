     <script>
          // ===== دریافت داده‌ها =====
          var questionsDataStr = '{{ questions_json }}';
          var durationStr = '{{ exam.duration }}';
          var baseUrl = '{{ base_url }}';

          var questionsData = JSON.parse(questionsDataStr);
          var duration = parseInt(durationStr, 10) || 0;
          var timeLeft = duration * 60;

          // ===== رندر سوالات =====
          function renderQuestions(questions) {
               var container = document.getElementById('questions-container');
               if (!container) return;
               if (!questions || questions.length === 0) {
                    container.innerHTML = '<p style="text-align:center;">هیچ سوالی برای این آزمون تعریف نشده است.</p>';
                    return;
               }

               var html = '';
               for (var i = 0; i < questions.length; i++) {
                    var q = questions[i];
                    var qNum = i + 1;
                    html += '<div class="question-container">';
                    html += '<h4>سوال ' + qNum + ': ' + (q.question_text || '') + '</h4>';
                    html += '<p><small>نمره: ' + (q.score || 0) + '</small></p>';

                    if (q.question_type === 'single' || q.question_type === 'multiple') {
                         var inputType = (q.question_type === 'single') ? 'radio' : 'checkbox';
                         var answers = q.answers || [];
                         for (var j = 0; j < answers.length; j++) {
                         var ans = answers[j];
                         html += '<label class="option-label" for="q' + q.id + '_' + ans.id + '">';
                         html += '<input type="' + inputType + '" id="q' + q.id + '_' + ans.id + '" ';
                         html += 'name="question_' + q.id + '" value="' + ans.id + '">';
                         html += '<span>' + ans.answer_text + '</span>';
                         html += '</label>';
                         }
                    } else {
                         html += '<div class="mdl-textfield mdl-js-textfield" style="width: 100%;">';
                         html += '<textarea class="mdl-textfield__input" id="q' + q.id + '" name="question_' + q.id + '" rows="3"></textarea>';
                         html += '<label class="mdl-textfield__label" for="q' + q.id + '">پاسخ خود را وارد کنید...</label>';
                         html += '</div>';
                    }

                    html += '</div>';
               }
               container.innerHTML = html;
          }

          // ===== تایمر =====
          function startTimer(totalSeconds) {
               var display = document.getElementById('time-display');
               if (!display) return;
               var timerInterval = setInterval(function() {
                    var minutes = Math.floor(totalSeconds / 60);
                    var seconds = totalSeconds % 60;
                    display.textContent = String(minutes).padStart(2, '0') + ':' + String(seconds).padStart(2, '0');
                    if (totalSeconds <= 0) {
                         clearInterval(timerInterval);
                         alert('زمان شما به پایان رسید!');
                         document.getElementById('exam-form').submit();
                    }
                    totalSeconds--;
               }, 1000);
          }

          // ===== بارگذاری صفحه =====
          window.onload = function() {
               renderQuestions(questionsData);
               if (duration > 0) startTimer(duration * 60);

               // جلوگیری از خروج تصادفی
               window.addEventListener('beforeunload', function(e) {
                    e.preventDefault();
                    e.returnValue = '';
               });

               if (window.componentHandler) {
                    window.componentHandler.upgradeAllRegistered();
               }
          };
     </script>
